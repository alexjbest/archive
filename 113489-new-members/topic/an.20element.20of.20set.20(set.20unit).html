instance set.fintype [fintype α] : fintype (set α) :=
⟨(@finset.univ α _).powerset.map ⟨coe, to_set_injective⟩, λ s,
  by classical; simp [subset_univ]; exact ⟨finset.univ.filter s, by simp; refl⟩⟩
