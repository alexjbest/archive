theorem wlog_is_slow (x y : ℕ) (h1 : x ^ y = 3) (h2 : x ^ y = 3) (h3 : x ^ y = 3) (h4 : x ^ y = 3) :
  ∃ m n, x = m ∧ y = n ∧ x + y = m + n ∧ x * y = m * n ∧ x ^ y = m ^ n :=
begin
  wlog hxy : x ≤ y := le_total x y using [x y, y x],
  exact ⟨x, y, rfl, rfl, rfl, rfl, rfl⟩
